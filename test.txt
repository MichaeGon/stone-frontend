def fact (n) {
    f = 1
    while n > 0 {
        f = f * n
        n = n - 1
    }
    f
}
fact(9)

class Position {
    x = y = 0
    def move (nx, ny) {
        x = nx
        y = ny
    }
}

p = Position.new
p.move(3, 4)
p.x = 10

class Pos3D extends Position {
    z = 0
    def set (nx, ny, nz) {
        x = nx
        y = ny
        z = nz
    }
}

p = Pos3D.new
p.move(3, 4)
p.x
p.set(5, 6, 7)
p.z
