def fact (n) {
    var f = 1
    while n > 0 {
        f = f * n
        n = n - 1
    }
    f
}
fact(9)

class Position {
    var x = 0
    var y = 0
    def move (nx, ny) {
        x = nx
        y = ny
    }
}
